Aquí está el prompt adaptado para un restaurante, manteniendo toda la lógica exacta:

🍽️ SISTEMA DE COSTEO — RESTAURANTE
🛠️ STACK TECNOLÓGICO
Backend: Python (Flask)
Base de datos: SQLite
Frontend: HTML + Jinja2 + CSS vanilla
⚠️ Sin React, sin Node, sin frameworks externos

🎯 OBJETIVO
Calcular el costo unitario real de cada plato del menú y generar un Estado de Resultados Unitario por plato, basado en: Materia Prima, MOD, CIF (incluyendo Depreciación), Gastos Administrativos, Gastos de Ventas, Gastos Financieros, Utilidad deseada e IGV (10%).

🔑 LÓGICA CENTRAL DEL SISTEMA
Paso 1 — Proyección de Demanda (origen de todo)
El usuario ingresa por plato: nombre, cantidad mensual y precio referencial.
El sistema calcula automáticamente:
ingreso_plato     = cantidad_plato × precio_referencial
total_cantidad    = Σ cantidad de todos los platos
total_ingresos    = Σ ingreso de todos los platos
pct_participacion = cantidad_plato / total_cantidad   ← PORCENTAJE CLAVE
⚠️ Este pct_participacion es el único factor para distribuir CIF (incluyendo depreciación), gastos administrativos, gastos de ventas y gastos financieros. Se recalcula automáticamente cada vez que cambia cualquier dato de proyección.
Paso 2 — Materia Prima (MP) con receta
Por cada ingrediente en la receta de un plato:
costo_unitario_real     = costo_compra / equivalencia
costo_total_ingrediente = costo_unitario_real × cantidad_uso
MP_plato                = Σ costo_total_ingrediente
La tabla de receta muestra: Ingrediente | Cant. usada | U. uso | C. de compra | U. compra | Costo total
❌ NO mostrar costo_unitario_real como columna separada.
Paso 3 — Mano de Obra Directa (MOD)
3a. sueldo_total_empleado = sueldo_base + gratificaciones + bonificaciones + seguro + CTS
3b. planilla_total = Σ sueldo_total_empleado
3c. tiempo_base_minutos = dias_laborables × horas_por_dia × 60
3d. tiempo_efectivo = tiempo_base_minutos × (productividad_pct / 100)
3e. costo_por_minuto = planilla_total / tiempo_efectivo
3f. costo_MOD_plato = costo_por_minuto × minutos_por_plato
⚠️ Mostrar costo_por_minuto = S/ X.XXXX visible en pantalla de Productividad como verificación.
Validaciones:

productividad_pct entre 1 y 100 (entero)
Si tiempo_efectivo = 0 → error: "No se puede calcular: tiempo efectivo es cero"
Si planilla_total = 0 → advertencia: "No hay empleados registrados"

Paso 4 — Costos Indirectos de Fabricación (CIF)
El CIF se compone de dos fuentes: conceptos manuales + depreciación de activos fijos.
4a. CIF_concepto_plato = (monto_concepto × pct_participacion) / cantidad_mensual
4b. dep_anual_activo   = (valor_adquisicion - valor_residual) / vida_util_años
    dep_mensual_activo = dep_anual_activo / 12
    total_dep_mensual  = Σ dep_mensual_activo   (solo activos con activo = 1)
    depreciacion_plato = (total_dep_mensual × pct_participacion) / cantidad_mensual
4c. CIF_total_plato = Σ CIF_concepto_plato + depreciacion_plato
Paso 5 — Gastos Administrativos
GA_concepto_plato = (monto_concepto × pct_participacion) / cantidad_mensual
GA_total_plato    = Σ GA_concepto_plato
Paso 6 — Gastos de Ventas
GV_concepto_plato = (monto_concepto × pct_participacion) / cantidad_mensual
GV_total_plato    = Σ GV_concepto_plato
Paso 7 — Gastos Financieros
GF_concepto_plato = (monto_concepto × pct_participacion) / cantidad_mensual
GF_total_plato    = Σ GF_concepto_plato
Paso 8 — Estado de Resultados Unitario
costo_produccion = MP_plato + MOD_plato + CIF_total_plato
costo_total      = costo_produccion + GA_total_plato + GV_total_plato + GF_total_plato
utilidad_S/      = costo_total × (utilidad_pct / 100)
valor_venta      = costo_total + utilidad_S/
IGV              = valor_venta × 0.10
precio_final     = valor_venta × 1.10

sqlCREATE TABLE platos (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  nombre TEXT NOT NULL,
  descripcion TEXT
);

CREATE TABLE ingredientes (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  nombre TEXT NOT NULL,
  costo_compra REAL NOT NULL,
  unidad_compra TEXT NOT NULL
);

CREATE TABLE conversiones (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  ingrediente_id INTEGER NOT NULL,
  equivalencia REAL NOT NULL,
  unidad_uso TEXT NOT NULL,
  FOREIGN KEY (ingrediente_id) REFERENCES ingredientes(id)
);

CREATE TABLE recetas (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  plato_id INTEGER NOT NULL,
  ingrediente_id INTEGER NOT NULL,
  cantidad_uso REAL NOT NULL,
  unidad_uso TEXT NOT NULL,
  FOREIGN KEY (plato_id) REFERENCES platos(id),
  FOREIGN KEY (ingrediente_id) REFERENCES ingredientes(id)
);

CREATE TABLE kardex (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  ingrediente_id INTEGER NOT NULL,

  cantidad REAL NOT NULL,
  unidad TEXT NOT NULL,
  costo_unitario REAL NOT NULL,
  costo_total REAL NOT NULL,
  saldo_cantidad REAL NOT NULL,
  saldo_valor REAL NOT NULL,
  FOREIGN KEY (ingrediente_id) REFERENCES ingredientes(id)
);

CREATE TABLE empleados (
  id INTEGER PRIMARY KEY AUTOINCREMENT,

python# ── Depreciación ──────────────────────────────────────────────
def calcular_depreciacion_mensual(db):
    activos = db.execute(
        "SELECT * FROM activos_fijos WHERE activo = 1"
    ).fetchall()
    total = 0
    detalle = []
    for a in activos:
        dep_anual   = (a['valor_adquisicion'] - a['valor_residual']) / a['vida_util_años']
        dep_mensual = dep_anual / 12
        total += dep_mensual
        detalle.append({**dict(a), 'dep_anual': dep_anual, 'dep_mensual': dep_mensual})
    return round(total, 4), detalle

def depreciacion_por_plato(db, pct_participacion, cantidad_mensual):
    total_dep, _ = calcular_depreciacion_mensual(db)
    if cantidad_mensual == 0:
        return 0
    return round((total_dep * pct_participacion) / cantidad_mensual, 4)

# ── Porcentaje de participación ───────────────────────────────
def calcular_pct_participacion(db):
    rows = db.execute(
        "SELECT plato_id, cantidad_mensual FROM proyeccion"
    ).fetchall()
    total_cantidad = sum(r['cantidad_mensual'] for r in rows)
    if total_cantidad == 0:
        return {}
    return {
        r['plato_id']: r['cantidad_mensual'] / total_cantidad
        for r in rows
    }

# ── CIF total por plato (manual + depreciación) ───────────────
def cif_por_plato(db, plato_id):
    pcts = calcular_pct_participacion(db)
    pct  = pcts.get(plato_id, 0)
    cant = db.execute(
        "SELECT cantidad_mensual FROM proyeccion WHERE plato_id = ?", (plato_id,)
    ).fetchone()
    if not cant or cant['cantidad_mensual'] == 0:
        return 0, []
    cantidad = cant['cantidad_mensual']
    conceptos = db.execute("SELECT * FROM costos_indirectos").fetchall()
    detalle = []
    total = 0
    for c in conceptos:
        valor = round((c['monto'] * pct) / cantidad, 4)
        total += valor
        detalle.append({'concepto': c['concepto'], 'monto': c['monto'], 'valor_plato': valor})
    dep = depreciacion_por_plato(db, pct, cantidad)
    detalle.append({'concepto': 'Depreciación activos', 'monto': None, 'valor_plato': dep})
    total += dep
    return round(total, 4), detalle

# ── MOD ───────────────────────────────────────────────────────
def mod_por_plato(db, plato_id):
    prod = db.execute(
        "SELECT * FROM produccion_empleado WHERE plato_id = ?", (plato_id,)
    ).fetchone()
    if not prod:
        return 0, 0
    planilla = db.execute(
        "SELECT SUM(sueldo_base+gratificaciones+bonificaciones+seguro+cts) AS total FROM empleados"
    ).fetchone()['total'] or 0
    tiempo_base     = prod['dias_laborables'] * prod['horas_por_dia'] * 60
    tiempo_efectivo = tiempo_base * (prod['productividad'] / 100)
    if tiempo_efectivo == 0:
        return 0, 0
    costo_minuto = round(planilla / tiempo_efectivo, 6)
    mod          = round(costo_minuto * prod['minutos_por_plato'], 4)
    return mod, costo_minuto
